```python
from sqlalchemy import Column, Integer, String, Float, Text, DateTime, ForeignKey
from sqlalchemy.orm import relationship
from datetime import datetime, timezone
from database import Base


class Prenotazione(Base):
    __tablename__ = "prenotazioni"

    id = Column(Integer, primary_key=True, index=True)
    tipo_richiesta = Column(String, default="Contatto")
    nome = Column(String, nullable=True)
    cognome = Column(String, nullable=True)
    telefono = Column(String, nullable=True)
    email = Column(String, nullable=True)
    data_arrivo = Column(String, nullable=True)
    data_partenza = Column(String, nullable=True)
    adulti = Column(Integer, nullable=True)
    bambini = Column(Integer, nullable=True)
    posto_per = Column(String, nullable=True)
    stato = Column(String, default="Nuova")
    costo_totale = Column(Float, nullable=True)
    data_ricezione = Column(DateTime, default=lambda: datetime.now(timezone.utc))
    message_id = Column(String, unique=True, nullable=True)
    lingua_suggerita = Column(String, nullable=True)

    messaggi = relationship(
        "StoricoMessaggio",
        back_populates="prenotazione",
        cascade="all, delete-orphan",
        order_by="StoricoMessaggio.data_ora",
    )


class StoricoMessaggio(Base):
    __tablename__ = "storico_messaggi"

    id = Column(Integer, primary_key=True, index=True)
    id_prenotazione = Column(
        Integer, ForeignKey("prenotazioni.id", ondelete="CASCADE"), nullable=False
    )
    mittente = Column(String, nullable=False)  # "Cliente" or "Campeggio"
    testo = Column(Text, nullable=True)
    testo_tradotto = Column(Text, nullable=True)
    data_ora = Column(DateTime, default=lambda: datetime.now(timezone.utc))
    message_id = Column(String, unique=True, nullable=True)

    prenotazione = relationship("Prenotazione", back_populates="messaggi")


class Impostazione(Base):
    __tablename__ = "impostazioni"

    id = Column(Integer, primary_key=True, index=True)
    chiave = Column(String, unique=True, nullable=False)
    valore = Column(Text, nullable=True)


class ModelloMail(Base):
    __tablename__ = "modelli_mail"

    id = Column(Integer, primary_key=True, index=True)
    lingua = Column(String, nullable=False)  # IT/EN/DE/FR/NL
    tipo = Column(String, nullable=False)  # accetta/rifiuta/info
    soggetto = Column(String, nullable=True)
    corpo = Column(Text, nullable=True)
```

```python
# ── Migration helper ──────────────────────────────────────────────

_MIGRATIONS = [
    ("prenotazioni", "lingua_suggerita", "VARCHAR"),
    ("prenotazioni", "costo_totale", "FLOAT"),
    ("storico_messaggi", "testo_tradotto", "TEXT"),

def _run_migrations():
    with engine.connect() as conn:
        for table, column, col_type in _MIGRATIONS:
            try:
                conn.execute(text(f"ALTER TABLE {table} ADD COLUMN {column} {col_type}"))
                conn.commit()
            except Exception:
                conn.rollback()


# ── Seed data ─────────────────────────────────────────────────────

def _seed_defaults():
    db = SessionLocal()
    try:
        # Settings
        for key, value in _DEFAULT_SETTINGS.items():
            existing = db.query(Impostazione).filter(Impostazione.chiave == key).first()
            if not existing:
                db.add(Impostazione(chiave=key, valore=value))

        # Price list
        existing_pl = db.query(Impostazione).filter(Impostazione.chiave == "listino_prezzi").first()
        if not existing_pl:
            db.add(Impostazione(chiave="listino_prezzi", valore=json.dumps(_DEFAULT_PRICE_LIST)))

        # Email templates
        for lingua, tipo, soggetto, corpo in _DEFAULT_TEMPLATES:
            existing_t = (
                db.query(ModelloMail)
                .filter(ModelloMail.lingua == lingua, ModelloMail.tipo == tipo)
                .first()
            )
            if not existing_t:
                db.add(ModelloMail(lingua=lingua, tipo=tipo, soggetto=soggetto, corpo=corpo))

        db.commit()
    except Exception:
        db.rollback()
    finally:
        db.close()


# ── Scheduler ─────────────────────────────────────────────────────

scheduler = BackgroundScheduler()


def _scheduled_poll():
    from services.mail_poller import poll_emails

    db = SessionLocal()
    try:
        poll_emails(db)
    except Exception:
        pass
    finally:
        db.close()


def _start_scheduler():
    db = SessionLocal()
    try:
        row = db.query(Impostazione).filter(Impostazione.chiave == "poll_interval_minutes").first()
        interval = int(row.valore) if row and row.valore else 10
    finally:
        db.close()

    if scheduler.get_job("poll_emails"):
        scheduler.remove_job("poll_emails")
    scheduler.add_job(_scheduled_poll, "interval", minutes=interval, id="poll_emails")
    if not scheduler.running:
        scheduler.start()


# ── App ───────────────────────────────────────────────────────────

@asynccontextmanager
async def lifespan(app: FastAPI):
    Base.metadata.create_all(bind=engine)
    _run_migrations()
    _seed_defaults()
    _start_scheduler()
    yield
    scheduler.shutdown(wait=False)


app = FastAPI(title="CampingV2", lifespan=lifespan)
```

```python
def _str_or_none(val):
    """Guard against Ollama returning list/dict instead of string."""
    if val is None:
        return None
    if isinstance(val, str):
        return val.strip() or None
    if isinstance(val, list):
        return str(val[0]).strip() if val else None
    return str(val).strip() or None


def _int_or_none(val):
    """Guard against Ollama returning string/list instead of int."""
    if val is None:
        return None
    if isinstance(val, int):
        return val
    if isinstance(val, float):
        return int(val)
    if isinstance(val, str):
        try:
            return int(val)
        except ValueError:
            return None
    if isinstance(val, list):
        return _int_or_none(val[0]) if val else None
    return None


def _float_or_none(val):
    if val is None:
        return None
    if isinstance(val, (int, float)):
        return float(val)
```

```python
import smtplib
from email.mime.text import MIMEText
from email.mime.multipart import MIMEMultipart
from email.utils import formatdate, make_msgid


def send_email(
    to_addr: str,
    subject: str,
    body: str,
    settings: dict,
    reply_to_message_id: str | None = None,
) -> str:
    """Send email via SMTP. Returns the new Message-ID.

    Args:
        to_addr: recipient email
        subject: email subject
        body: email body (plain text)
        settings: dict of Impostazione key-value pairs
        reply_to_message_id: Message-ID to set In-Reply-To header (for threading)

    Raises:
        Exception on SMTP failure
    """
    smtp_server = settings.get("smtp_server", "smtps.aruba.it")
    smtp_port = int(settings.get("smtp_port", "587"))
    smtp_user = settings.get("smtp_user", "")
    smtp_password = settings.get("smtp_password", "")
    from_addr = settings.get("email_mittente", smtp_user)

    msg = MIMEMultipart()
    msg["From"] = from_addr
    msg["To"] = to_addr
    msg["Subject"] = subject
    msg["Date"] = formatdate(localtime=True)
    new_message_id = make_msgid()
    msg["Message-ID"] = new_message_id

    if reply_to_message_id:
        msg["In-Reply-To"] = reply_to_message_id
        msg["References"] = reply_to_message_id

    msg.attach(MIMEText(body, "plain", "utf-8"))

    with smtplib.SMTP(smtp_server, smtp_port) as server:
        server.starttls()
        server.login(smtp_user, smtp_password)
        server.send_message(msg)

    return new_message_id
```

```python
def _decode_header(raw: str | None) -> str:

def _discard(from_addr: str, subject: str, settings: dict) -> bool:
    """Level 1 spam filter: prefix/domain/subject blacklists. Returns True to discard."""
    addr_lower = from_addr.lower()
    local_part = addr_lower.split("@")[0] if "@" in addr_lower else addr_lower

    # Prefix blacklist
    for prefix in _NOREPLY_PREFIXES:
        if local_part.startswith(prefix) or local_part == prefix:
            return True

    # Domain keyword blacklist (built-in)
    for kw in _SPAM_DOMAIN_KEYWORDS:
        if kw in addr_lower:
            return True

    # User-configurable domain blacklist
    user_domains = settings.get("filtro_domini_scarta", "")
    if user_domains:
        for frag in user_domains.split(","):
            frag = frag.strip().lower()
            if frag and frag in addr_lower:
                return True

    # User-configurable subject blacklist
    user_subjects = settings.get("filtro_oggetto_scarta", "")
    if user_subjects:
        subj_lower = subject.lower()
        for frag in user_subjects.split(","):
            frag = frag.strip().lower()
            if frag and frag in subj_lower:

    return False

def _find_sent_folder(conn: imaplib.IMAP4_SSL) -> str | None:
    """Auto-detect Sent folder name from IMAP LIST response."""
    typ, data = conn.list()
    if typ != "OK":
        return None
    sent_patterns = re.compile(
        r"(sent|inviata|envoy|verzonden|gesendet)", re.IGNORECASE
    )
    for item in data:
        if isinstance(item, bytes):
            decoded = item.decode("utf-8", errors="replace")
            # Extract folder name (last part after delimiter)
            match = re.search(r'"[^"]*"\s+"?([^"]+)"?\s*$', decoded)
            if not match:
                match = re.search(r'\)\s+"."\s+"?([^"]+)"?\s*$', decoded)
            if match:
                folder = match.group(1).strip('"').strip()
                if sent_patterns.search(folder):
                    return folder
    return None


def _batch_fetch_headers(conn: imaplib.IMAP4_SSL, folder: str, limit: int = 0) -> dict:
    """Fetch email headers from folder in batches of 500.

    Returns dict: {message_id: {from, to, date, subject, message_id, in_reply_to, references, folder, uid}}
    """
    try:
        typ, _ = conn.select(folder, readonly=True)
        if typ != "OK":
            return {}
    except Exception:
        return {}

    typ, data = conn.search(None, "ALL")
    if typ != "OK" or not data[0]:
        return {}

    uids = data[0].split()
    if limit > 0:
        uids = uids[-limit:]

    headers = {}
    batch_size = 500
    header_fields = "(BODY.PEEK[HEADER.FIELDS (FROM TO DATE SUBJECT MESSAGE-ID IN-REPLY-TO REFERENCES)])"

    for i in range(0, len(uids), batch_size):
        batch = uids[i : i + batch_size]
        uid_range = b",".join(batch)
        try:
            typ, data = conn.fetch(uid_range, header_fields)
        except Exception:
            continue
        if typ != "OK":
            continue

        current_uid = None
        for item in data:
            if isinstance(item, tuple) and len(item) >= 2:
                # Extract UID from response
                uid_match = re.search(rb"(\d+)\s+\(", item[0])
                if uid_match:
                    current_uid = uid_match.group(1)
                raw_headers = item[1]
                if isinstance(raw_headers, bytes):
                    msg = email.message_from_bytes(raw_headers)
                    mid = _clean_message_id(msg.get("Message-ID"))
                    if not mid:
                        continue
                    headers[mid] = {
                        "from": _extract_email_addr(msg.get("From")),
                        "from_raw": _decode_header(msg.get("From")),
                        "to": _extract_email_addr(msg.get("To")),
                        "date": msg.get("Date"),
                        "subject": _decode_header(msg.get("Subject")),
                        "message_id": mid,
                        "in_reply_to": _clean_message_id(msg.get("In-Reply-To")),
                        "references": msg.get("References", ""),
                        "folder": folder,
                        "uid": current_uid,
                    }
    return headers


def _fetch_body(conn: imaplib.IMAP4_SSL, uid: bytes, folder: str) -> str:
    """Fetch and extract email body text, switching to correct folder."""
    try:
        conn.select(folder, readonly=True)
        typ, data = conn.fetch(uid, "(BODY[])")
        if typ != "OK" or not data or not data[0]:
            return ""
    except Exception:
        return ""

    raw = data[0][1] if isinstance(data[0], tuple) else data[0]
    if not isinstance(raw, bytes):
        return ""

    msg = email.message_from_bytes(raw)
    text_parts = []

    if msg.is_multipart():
        for part in msg.walk():
            ct = part.get_content_type()
            if ct == "text/plain":
                payload = part.get_payload(decode=True)
                if payload:
                    charset = part.get_content_charset() or "utf-8"
                    text_parts.append(payload.decode(charset, errors="replace"))
            elif ct == "text/html" and not text_parts:
                payload = part.get_payload(decode=True)
                if payload:
                    charset = part.get_content_charset() or "utf-8"
                    text_parts.append(_strip_html(payload.decode(charset, errors="replace")))
    else:
        payload = msg.get_payload(decode=True)
        if payload:
            charset = msg.get_content_charset() or "utf-8"
            text = payload.decode(charset, errors="replace")
            if msg.get_content_type() == "text/html":
                text = _strip_html(text)
            text_parts.append(text)

    full_text = "\n".join(text_parts).strip()
    return _strip_quoted_text(full_text)


# ── Thread reconstruction ─────────────────────────────────────────


def _find_root(msg_id: str, header_map: dict) -> str:
    """Walk References/In-Reply-To chain to find thread root message ID."""
    visited = set()
    current = msg_id

    while current not in visited:
        visited.add(current)
        info = header_map.get(current)
        if not info:
            break

        # Check References header (first entry = original message)
        refs_raw = info.get("references", "")
        if refs_raw:
            ref_ids = [
                _clean_message_id(r)
                for r in re.split(r"\s+", refs_raw.strip())
                if _clean_message_id(r)
            ]
            if ref_ids and ref_ids[0] in header_map:
                current = ref_ids[0]
                continue

        # Fallback: In-Reply-To
        reply_to = info.get("in_reply_to")
        if reply_to and reply_to in header_map:
            current = reply_to
            continue

        break

    return current


def _group_threads(header_map: dict) -> dict[str, list[str]]:
    """Group message IDs into threads. Returns {root_id: [msg_id1, msg_id2, ...]}."""
    threads = {}
    msg_to_root = {}

    for mid in header_map:
        root = _find_root(mid, header_map)
        msg_to_root[mid] = root
        threads.setdefault(root, []).append(mid)

    # Secondary fallback: group by normalized subject + sender for orphans
    # (messages whose root is themselves and have no References)
    orphans = {}
    for root_id, members in list(threads.items()):
        if len(members) == 1:
            info = header_map[root_id]
            key = (_normalize_subject(info["subject"]), info["from"])
            orphans.setdefault(key, []).append(root_id)

    # Merge orphan groups that share subject+sender
    for key, root_ids in orphans.items():
        if len(root_ids) > 1:
            primary = root_ids[0]
            for other in root_ids[1:]:
                threads[primary].extend(threads.pop(other, []))

    # Sort each thread chronologically
    for root_id in threads:
        threads[root_id].sort(
            key=lambda mid: _parse_date(header_map[mid]["date"])
        )

    return threads

def _determine_mittente(from_addr: str, settings: dict) -> str:
    """Determine if email is from campsite or client."""
    campsite_emails = [
        settings.get("email_mittente", "").lower(),
        settings.get("imap_user", "").lower(),
        settings.get("email_form_sito", "").lower(),
    ]
    return "Campeggio" if from_addr.lower() in campsite_emails else "Cliente"


def _process_thread_with_ollama(
    root_id: str,
    thread_msgs: list[str],
    header_map: dict,
    bodies: dict,
    settings: dict,
) -> dict | None:
    """Classify a thread via Ollama. Returns parsed data or None if spam.
    
    Runs in ThreadPoolExecutor worker — uses own DB session.
    """
    first_msg = header_map[root_id]
    body = bodies.get(root_id, "")
    if not body:
        # Try first message with a body
        for mid in thread_msgs:
            if mid in bodies and bodies[mid]:
                body = bodies[mid]
                break

    parsed = parse_email(first_msg["from"], first_msg["subject"], body, settings)

    # Discard spam
    if parsed["tipo"] == "spam" and parsed["confidenza"] >= 0.5:
        return None

    return parsed


def _save_thread(
    db,
    root_id: str,
    thread_msgs: list[str],
    header_map: dict,
    bodies: dict,
    parsed: dict,
    settings: dict,
):
    """Save a thread as Prenotazione + StoricoMessaggio rows."""
    first_msg = header_map[root_id]

    pren = Prenotazione(
        tipo_richiesta=parsed.get("tipo", "Contatto").capitalize()
        if parsed.get("tipo") != "spam"
        else "Contatto",
        nome=parsed.get("nome"),
        cognome=parsed.get("cognome"),
        telefono=parsed.get("telefono"),
        email=first_msg["from"],
        data_arrivo=parsed.get("data_arrivo"),
        data_partenza=parsed.get("data_partenza"),
        adulti=parsed.get("adulti"),
        bambini=parsed.get("bambini"),
        posto_per=parsed.get("posto_per"),
        stato="Nuova",
        data_ricezione=_parse_date(first_msg["date"]),
        message_id=root_id,
        lingua_suggerita=parsed.get("lingua"),
    )
    # Normalize tipo_richiesta
    if pren.tipo_richiesta and pren.tipo_richiesta.lower() == "prenotazione":
        pren.tipo_richiesta = "Prenotazione"
    else:
        pren.tipo_richiesta = "Contatto"

    db.add(pren)
    db.flush()

    for mid in thread_msgs:
        info = header_map.get(mid)
        if not info:
            continue
        # Skip if message already in DB
        existing = (
            db.query(StoricoMessaggio)
            .filter(StoricoMessaggio.message_id == mid)
            .first()
        )
        if existing:
            continue

        mittente = _determine_mittente(info["from"], settings)
        body = bodies.get(mid, "")

        sm = StoricoMessaggio(
            id_prenotazione=pren.id,
            mittente=mittente,
            testo=body,
            data_ora=_parse_date(info["date"]),
            message_id=mid,
        )
        db.add(sm)

    db.commit()


def _append_to_thread(
    db,
    pren: Prenotazione,
    msg_ids: list[str],
    header_map: dict,
    bodies: dict,
    settings: dict,
):
    """Append new messages to an existing Prenotazione thread."""
    added = False
    for mid in msg_ids:
        existing = (
            db.query(StoricoMessaggio)
            .filter(StoricoMessaggio.message_id == mid)
            .first()
        )
        if existing:
            continue

        info = header_map.get(mid)
        if not info:
            continue

        mittente = _determine_mittente(info["from"], settings)
        body = bodies.get(mid, "")

        sm = StoricoMessaggio(

            mittente=mittente,
            testo=body,
            data_ora=_parse_date(info["date"]),
            message_id=mid,

        if mittente == "Cliente" and pren.stato not in ("Nuova", "Nuova Risposta"):
            pren.stato = "Nuova Risposta"

    if added:
        db.commit()

def poll_emails(db, limit: int = 20):
    """Quick poll: check last N emails in INBOX. Synchronous."""
    settings = _load_settings(db)
    conn = _connect_imap(settings)

    try:
        inbox_headers = _batch_fetch_headers(conn, "INBOX", limit=limit)
        sent_folder = _find_sent_folder(conn)
        sent_headers = {}
        if sent_folder:
            sent_headers = _batch_fetch_headers(conn, sent_folder, limit=limit)

        all_headers = {**sent_headers, **inbox_headers}
        threads = _group_threads(all_headers)

        for root_id, msg_ids in threads.items():
            first_msg = all_headers[root_id]

            # Skip if already fully imported
            existing_pren = (
                db.query(Prenotazione)
                .filter(Prenotazione.message_id == root_id)
                .first()
            )

            # Level 1: spam filter
            if _discard(first_msg["from"], first_msg["subject"], settings):
                continue

            if existing_pren:
                # Level 2: known thread — append new messages
                bodies = {}
                for mid in msg_ids:
                    existing_sm = (
                        db.query(StoricoMessaggio)
                        .filter(StoricoMessaggio.message_id == mid)
                        .first()
                    )
                    if not existing_sm:
                        bodies[mid] = _fetch_body(
                            conn, all_headers[mid]["uid"], all_headers[mid]["folder"]
                        )
                _append_to_thread(db, existing_pren, msg_ids, all_headers, bodies, settings)
                continue

            # Level 2: known client by email
            known_pren = (
                db.query(Prenotazione)
                .filter(Prenotazione.email == first_msg["from"])
                .first()
            )
            if known_pren:
                bodies = {
                    mid: _fetch_body(conn, all_headers[mid]["uid"], all_headers[mid]["folder"])
                    for mid in msg_ids
                }
                _append_to_thread(db, known_pren, msg_ids, all_headers, bodies, settings)
                continue

            # Level 3: Ollama classification
            try:
                bodies = {
                    mid: _fetch_body(conn, all_headers[mid]["uid"], all_headers[mid]["folder"])
                    for mid in msg_ids
                }
                parsed = _process_thread_with_ollama(
                    root_id, msg_ids, all_headers, bodies, settings
                )
                if parsed:
                    _save_thread(db, root_id, msg_ids, all_headers, bodies, parsed, settings)
            except Exception as e:
                db.rollback()
                logger.error(f"Error processing thread {root_id}: {e}")

    finally:
        try:
            conn.close()
            conn.logout()
        except Exception:
            pass


def import_full_history(
    db=None,
    mail_limit: int = 0,
    ollama_limit: int = 100,
    job_state: dict | None = None,
):
    """Full history import. Runs in background thread.

    Args:
        db: if None, creates own session (for background thread use)
        mail_limit: max emails from INBOX (0 = all)
        ollama_limit: max threads to send to Ollama
        job_state: mutable dict for progress tracking
    """
    own_session = db is None
    if own_session:
        db = SessionLocal()

    try:
        settings = _load_settings(db)
        conn = _connect_imap(settings)

        try:
            # Fetch headers from INBOX
            inbox_headers = _batch_fetch_headers(conn, "INBOX", limit=mail_limit)
            if job_state:
                job_state["status"] = "fetching_sent"

            # Fetch headers from Sent
            sent_folder = _find_sent_folder(conn)
            sent_headers = {}
            if sent_folder:
                sent_headers = _batch_fetch_headers(conn, sent_folder, limit=0)

            all_headers = {**sent_headers, **inbox_headers}
            threads = _group_threads(all_headers)

            if job_state:
                job_state["total"] = len(threads)
                job_state["status"] = "processing"

            # Categorize threads
            needs_ollama = []
            ollama_count = 0

            for root_id, msg_ids in threads.items():
                first_msg = all_headers[root_id]

                # Skip already imported
                existing_pren = (
                    db.query(Prenotazione)
                    .filter(Prenotazione.message_id == root_id)
                    .first()
                )

                # Level 1: spam filter
                if _discard(first_msg["from"], first_msg["subject"], settings):
                    if job_state:
                        job_state["processed"] += 1
                    continue

                if existing_pren:
                    # Append new messages to existing thread
                    try:
                        bodies = {}
                        for mid in msg_ids:
                            existing_sm = (
                                db.query(StoricoMessaggio)
                                .filter(StoricoMessaggio.message_id == mid)
                                .first()
                            )
                            if not existing_sm:
                                bodies[mid] = _fetch_body(
                                    conn, all_headers[mid]["uid"], all_headers[mid]["folder"]
                                )
                        if bodies:
                            _append_to_thread(
                                db, existing_pren, msg_ids, all_headers, bodies, settings
                            )
                    except Exception as e:
                        db.rollback()
                        logger.error(f"Error appending to thread {root_id}: {e}")
                    if job_state:
                        job_state["processed"] += 1
                    continue

                # Level 2: known client
                known_pren = (
                    db.query(Prenotazione)
                    .filter(Prenotazione.email == first_msg["from"])
                    .first()
                )
                if known_pren:
                    try:
                        bodies = {
                            mid: _fetch_body(
                                conn, all_headers[mid]["uid"], all_headers[mid]["folder"]
                            )
                            for mid in msg_ids
                        }
                        _append_to_thread(
                            db, known_pren, msg_ids, all_headers, bodies, settings
                        )
                    except Exception as e:
                        db.rollback()
                        logger.error(f"Error appending known client {root_id}: {e}")
                    if job_state:
                        job_state["processed"] += 1
                    continue

                # Queue for Ollama (Level 3)
                if ollama_count < ollama_limit:
                    needs_ollama.append((root_id, msg_ids))
                    ollama_count += 1
                else:
                    if job_state:
                        job_state["processed"] += 1

            # Fetch bodies for Ollama queue (sequential — IMAP is single-threaded)
            all_bodies = {}
            if job_state:
                job_state["status"] = "fetching_bodies"
            for root_id, msg_ids in needs_ollama:
                for mid in msg_ids:
                    if mid not in all_bodies:
                        all_bodies[mid] = _fetch_body(
                            conn, all_headers[mid]["uid"], all_headers[mid]["folder"]
                        )

            # Close IMAP before Ollama (long-running, don't hold connection)
            try:
                conn.close()
                conn.logout()
            except Exception:
                pass
            conn = None

            # Process through Ollama in parallel
            if job_state:
                job_state["status"] = "classifying"
            n_workers = int(settings.get("ollama_workers", "4"))

            def _worker(item):
                root_id, msg_ids = item
                worker_db = SessionLocal()
                try:
                    parsed = _process_thread_with_ollama(
                        root_id, msg_ids, all_headers, all_bodies, settings
                    )
                    if parsed:
                        # Check one more time for duplicates
                        exists = (
                            worker_db.query(Prenotazione)
                            .filter(Prenotazione.message_id == root_id)
                            .first()
                        )
                        if not exists:
                            _save_thread(
                                worker_db, root_id, msg_ids, all_headers,
                                all_bodies, parsed, settings,
                            )
                    return root_id, True
                except Exception as e:
                    worker_db.rollback()
                    return root_id, str(e)
                finally:
                    worker_db.close()

            with ThreadPoolExecutor(max_workers=n_workers) as executor:
                futures = {executor.submit(_worker, item): item for item in needs_ollama}
                for future in as_completed(futures):
                    root_id, result = future.result()
                    if job_state:
                        job_state["processed"] += 1
                        if isinstance(result, str):
                            job_state["errors"].append(f"{root_id}: {result}")

        finally:
            if conn:
                try:
                    conn.close()
                    conn.logout()
                except Exception:
                    pass

        if job_state:
            job_state["status"] = "completed"

    except Exception as e:
        if job_state:
            job_state["status"] = "error"
            job_state["errors"].append(str(e))
        logger.error(f"Import failed: {e}")
    finally:
        if own_session:
            db.close()


def reset_and_reimport(
    db=None,
    ollama_limit: int = 100,
```

```python
from models import Impostazione

router = APIRouter()

# Module-level job state (single background job at a time)
_job_state = {
    "job_id": None,
    "status": "idle",
    "total": 0,
    "processed": 0,
    "errors": [],
    "started_at": None,
}
_job_lock = threading.Lock()


def _load_settings(db: Session) -> dict:
    rows = db.query(Impostazione).all()
    return {r.chiave: r.valore for r in rows}


def _reset_job_state():
    _job_state["job_id"] = str(uuid4())
    _job_state["status"] = "running"
    _job_state["total"] = 0
    _job_state["processed"] = 0
    _job_state["errors"] = []
    _job_state["started_at"] = datetime.now(timezone.utc).isoformat()


@router.post("/poll")
def mail_poll(limit: int = 20, db: Session = Depends(get_db)):
    """Quick poll: synchronous, checks last N emails."""
    poll_emails(db, limit=limit)
    return {"ok": True}


@router.post("/import-full")
def mail_import_full(
    background_tasks: BackgroundTasks,
    mail_limit: int = 0,
    ollama_limit: int = 100,
):
    """Full history import. Runs in background, returns immediately."""
    with _job_lock:
        if _job_state["status"] == "running":
            return {"error": "A job is already running", "job_id": _job_state["job_id"]}
        _reset_job_state()
        job_id = _job_state["job_id"]

    def _run():
        import_full_history(
            db=None,
            mail_limit=mail_limit,
            ollama_limit=ollama_limit,
            job_state=_job_state,
        )

    background_tasks.add_task(_run)
    return {"ok": True, "job_id": job_id, "status": "running"}


@router.post("/reset-reimport")
def mail_reset_reimport(
    background_tasks: BackgroundTasks,
    ollama_limit: int = 100,
):
    """Delete all and reimport. Runs in background."""
    with _job_lock:
        if _job_state["status"] == "running":
            return {"error": "A job is already running", "job_id": _job_state["job_id"]}
        _reset_job_state()
        job_id = _job_state["job_id"]

    def _run():
        reset_and_reimport(
            db=None,
            ollama_limit=ollama_limit,
            job_state=_job_state,
        )

    background_tasks.add_task(_run)
    return {"ok": True, "job_id": job_id, "status": "running"}
```

```javascript
import { defineStore } from 'pinia'
import {
  getPrenotazioni,
  getPrenotazione,
  getJobStatus,
} from '../api'

export const usePrenotazioniStore = defineStore('prenotazioni', {
  state: () => ({
    list: [],
    selected: null,
    filters: {
      stato: '',
      tipo: '',
      search: '',
    },
    jobStatus: null,
    _jobPollInterval: null,
  }),

  getters: {
    filtered(state) {
      let items = state.list

      if (state.filters.stato) {
        items = items.filter((p) => p.stato === state.filters.stato)
      }

      if (state.filters.tipo) {
        items = items.filter((p) => p.tipo_richiesta === state.filters.tipo)
      }

      if (state.filters.search) {
        const q = state.filters.search.toLowerCase()
        items = items.filter(
          (p) =>
            (p.nome && p.nome.toLowerCase().includes(q)) ||
            (p.cognome && p.cognome.toLowerCase().includes(q)) ||
            (p.email && p.email.toLowerCase().includes(q))
        )
      }

      return items
    },
  },

  actions: {
    async fetchAll() {
      const { data } = await getPrenotazioni()
      this.list = data
    },

    async selectPrenotazione(id) {
      if (!id) {
        this.selected = null
        return
      }
      const { data } = await getPrenotazione(id)
      this.selected = data
    },

    startJobPolling() {
      if (this._jobPollInterval) return
      this._jobPollInterval = setInterval(async () => {
        try {
          const { data } = await getJobStatus()
          this.jobStatus = data
          if (data.status === 'completed' || data.status === 'error' || data.status === 'idle') {
            this.stopJobPolling()
            this.fetchAll()
          }
        } catch {
          // ignore polling errors
        }
      }, 5000)
    },

    stopJobPolling() {
      if (this._jobPollInterval) {
        clearInterval(this._jobPollInterval)
        this._jobPollInterval = null
      }
    },
  },
})
```

## Architecture

**Stack:** FastAPI + SQLite (SQLAlchemy) backend, Vue 3 + Pinia + Vite + Tailwind CSS frontend. All config stored in `impostazioni` SQLite table as key-value pairs (no .env file).

**Frontend -> Backend:** Vite proxies `/api/*` to `http://localhost:8000`.

**Email pipeline** (`backend/services/mail_poller.py`):
1. Level 1 spam filter: prefix/domain/subject blacklists (instant discard)
2. Level 2 fast-path: known client email or existing thread -> append, skip Ollama
3. Level 3: Ollama classifies remaining emails as prenotazione/contatto/spam

**Thread reconstruction:** batch-fetch IMAP headers (500/batch) from INBOX + Sent, group by References/In-Reply-To chain, fallback to normalized subject + sender.

**Background jobs:** FastAPI BackgroundTasks + ThreadPoolExecutor for parallel Ollama. Frontend polls `/api/mail/job-status` every 5s.

**Booking status flow:** Nuova -> In lavorazione -> Attesa Bonifico -> Confermata (or -> Rifiutata). Client reply auto-sets "Nuova Risposta".
